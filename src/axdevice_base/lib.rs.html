30/// Represents the configuration of an emulated device for a virtual machine.
31#[derive(Debug, Default, Clone, serde::Serialize, serde::Deserialize)]
32pub struct EmulatedDeviceConfig {
33    /// The name of the device
34    pub name: String,
35    /// The base IPA (Intermediate Physical Address) of the device.
36    pub base_ipa: usize,
37    /// The length of the device.
38    pub length: usize,
39    /// The IRQ (Interrupt Request) ID of the device.
40    pub irq_id: usize,
41    /// The type of emulated device.
42    pub emu_type: usize,
43    /// The config_list of the device
44    pub cfg_list: Vec<usize>,
45}

47/// [`BaseDeviceOps`] is the trait that all emulated devices must implement.
48pub trait BaseDeviceOps<R: DeviceAddrRange>: Any {
49    /// Returns the type of the emulated device.
50    fn emu_type(&self) -> EmuDeviceType;
51    /// Returns the address range of the emulated device.
52    fn address_range(&self) -> R;
53    /// Handles a read operation on the emulated device.
54    fn handle_read(&self, addr: R::Addr, width: AccessWidth) -> AxResult<usize>;
55    /// Handles a write operation on the emulated device.
56    fn handle_write(&self, addr: R::Addr, width: AccessWidth, val: usize) -> AxResult;
57}
58
59/// Determines whether the given device is of type `T` and calls the provided function `f` with a
60/// reference to the device if it is.
61pub fn map_device_of_type<T: BaseDeviceOps<R>, R: DeviceAddrRange, U, F: FnOnce(&T) -> U>(
62    device: &Arc<dyn BaseDeviceOps<R>>,
63    f: F,
64) -> Option<U> {
65    let any_arc: Arc<dyn Any> = device.clone();
66
67    any_arc.downcast_ref::<T>().map(f)
68}

70// trait aliases are limited yet: https://github.com/rust-lang/rfcs/pull/3437
71/// [`BaseMmioDeviceOps`] is the trait that all emulated MMIO devices must implement.
72/// It is a trait alias of [`BaseDeviceOps`] with [`GuestPhysAddrRange`] as the address range.
73pub trait BaseMmioDeviceOps = BaseDeviceOps<GuestPhysAddrRange>;
74/// [`BaseSysRegDeviceOps`] is the trait that all emulated system register devices must implement.
75/// It is a trait alias of [`BaseDeviceOps`] with [`SysRegAddrRange`] as the address range.
76pub trait BaseSysRegDeviceOps = BaseDeviceOps<SysRegAddrRange>;
77/// [`BasePortDeviceOps`] is the trait that all emulated port devices must implement.
78/// It is a trait alias of [`BaseDeviceOps`] with [`PortRange`] as the address range.
79pub trait BasePortDeviceOps = BaseDeviceOps<PortRange>;
